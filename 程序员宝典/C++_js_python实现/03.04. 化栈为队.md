class MyQueue:

    def __init__(self):
        """
        Initialize your data structure here.
        """
        self.A = []
        self.B = []


    def push(self, x: int) -> None:
        """
        Push element x to the back of queue.
        """
        self.A.append(x)


    def pop(self) -> int:
        """
        Removes the element from in front of queue and returns that element.
        """
        if self.B: 
            return self.B.pop()
        else:
            while self.A:
                self.B.append(self.A.pop())

            if self.B:
                return self.B.pop();
            # 添加到B后，B还是空
            else:
                return -1


    def peek(self) -> int:
        """
        Get the front element.
        """
        if self.B: 
            return self.B[-1]
        else:
            while self.A:
                self.B.append(self.A.pop())

            if self.B:
                return self.B[-1];
            # 添加到B后，B还是空
            else:
                return -1


    def empty(self) -> bool:
        """
        Returns whether the queue is empty.
        """
        return len(self.A) == 0 and len(self.B) == 0
