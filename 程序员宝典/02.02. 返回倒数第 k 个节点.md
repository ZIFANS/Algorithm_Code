# Python 

```python
class Solution:
    def kthToLast(self, head: ListNode, k: int) -> int:
        if head == None:
            return head
        
        # slow, fast = head, head
        slow = fast = head

        for i in range(k):
            fast = fast.next

        while fast:
            slow = slow.next
            fast = fast.next
        
        return slow.val
```

```js
var kthToLast = function(head, k) {
    if (!head)
        return null;
    
    let fast = head, slow = head;

    for (let i = 0; i < k; ++i) {
        fast = fast.next;
    }

    while (fast) {
        slow = slow.next;
        fast = fast.next;
    }

    return slow.val;
};
```

# C++

```C++
// 1、快慢指针
class Solution {
public:
    int kthToLast(ListNode* head, int k) {
        ListNode *slow = head, *fast = head;

        for (int i = 0; i < k; ++i) {
            fast = fast->next;
        }

        while (fast) {
            slow = slow->next;
            fast = fast->next;
        }

        return slow->val;
    }
};
```
