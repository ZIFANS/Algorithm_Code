# Oracle

```sql
-- 1、用or，比union稍微慢一些
select a.name name, a.population population, a.area area
from World a 
where a.population > 25000000 or a.area > 3000000

-- 2、用union
select a.name name, a.population population, a.area area
from World a 
where a.population > 25000000 
union
select a.name name, a.population population, a.area area
from World a 
where a.area > 3000000
```
