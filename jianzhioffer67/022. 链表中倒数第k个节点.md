# Python 

```python
# 双指针
class Solution:
    def getKthFromEnd(self, head: ListNode, k: int) -> ListNode:
        if head == None:
            return head
        
        # slow, fast = head, head
        slow = fast = head

        for i in range(k):
            fast = fast.next

        while fast:
            slow = slow.next
            fast = fast.next
        
        return slow

```


# JavaScript

```js
// 1.第一想法就是快慢指针，这里遍历了2次
var getKthFromEnd = function(head, k) {
    if (!head || k < 0)
        return null;

    let slow = head, fast = head;

    for (let i = 0; i < k; ++i) {
        fast = fast.next;
    }

    while (fast) {
        slow = slow.next;
        fast = fast.next;
    }

    return slow;
};

// 2.也是双指针，slow, fast两个指针，让fast先走k步，然后slow, fast一起走，直到fast为null
// 但是只需要一次遍历
var getKthFromEnd = function(head, k) {
    let slow = head, fast = head;
    let count = 0;

    while (fast) {
        if (count >= k) {
            slow = slow.next;
        }
        ++count;
        fast = fast.next;
    }

    return count < k ? null : slow;
};

```

# C++

```C++
/*
题目描述：
输入一个链表，输出该链表中倒数第k个结点。
*/

/*
双指针
*/
class Solution {
public:
    ListNode* getKthFromEnd(ListNode* head, int k) {
        if (!head)
            return head;
        
        ListNode* slow = head, *fast = head;

        for (int i = 0; i < k; ++i) {
            fast = fast->next;
        }

        while (fast) {
            slow = slow->next;
            fast = fast->next;
        }

        return slow;
    }
};
```
