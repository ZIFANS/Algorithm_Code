        :rtype: str
        """

                q.append(temp.right)
            else:
                ans.append("null")

                q.append(temp.left)

            i += 1

            if vals[i] != "null":
                temp.right = TreeNode(int(vals[i]))
                q.append(temp.right)
            
            i += 1

        return root
