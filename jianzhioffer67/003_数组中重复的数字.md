# Python

```python
# 用Set
# 空间复杂度：O(N)
class Solution:
    def findRepeatNumber(self, nums: List[int]) -> int:
        dic = set()

        for num in nums:
            if num in dic:
                return num
            dic.add(num)
        

        return -1

```

var findRepeatNumber = function(nums) {
    let map = new Map();

    for (let i = 0; i < nums.length; ++i) {
        if (map.has(nums[i])) {
            return nums[i];
        }

        map.set(nums[i], true);
    }
};

// 1、hash
class Solution {
public:
    int findRepeatNumber(vector<int>& nums) {
        unordered_map<int, bool> hash;

        for (int i = 0; i < nums.size(); ++i) {
            if (hash[nums[i]])
                return nums[i];
            else
                hash[nums[i]] = true;
        }

        return -1;
    }
};
