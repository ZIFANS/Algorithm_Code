class Solution:
    def singleNonDuplicate(self, nums: List[int]) -> int:
        map1 = {}

        for i in range(len(nums)):
            if nums[i] in map1:
                map1[nums[i]] += 1
            else:
                map1[nums[i]] = 1
        
        for i in range(len(nums)):
            if (map1[nums[i]] == 1):
                return nums[i]
        
        return -1

// 1、时间、空间复杂度都达不到要求
var singleNonDuplicate = function(nums) {
    let map = new Map();

    for (let i = 0; i < nums.length; ++i) {
        if (!map.has(nums[i])) {
            map.set(nums[i], 1);
        } else {
            let temp = map.get(nums[i]) + 1;
            map.set(nums[i], temp);
        }
    }

    for (let i = 0; i < nums.length; ++i) {
        if (map.get(nums[i]) === 1)
            return nums[i];
    }

    return -1;
};

// 2、二分。这道题还要领会领会
var singleNonDuplicate = function(nums) {
    let left = 0, right = nums.length;
    let L, R;

    while (left <= right) {
        let mid = left + ((right - left) >> 1);

        if (nums[mid] === nums[mid + 1]) {
            L = mid;
            R = mid + 1;
        } else if (nums[mid] === nums[mid - 1]) {
            L = mid - 1;
            R = mid;
        } else {
            return nums[mid];
        }

        if (L % 2 === 0 && R % 2 === 1)
            left = mid + 1;
        else 
            right = mid - 1;
    }

    return -1;
};

// 2、二分的另外一种写法
var singleNonDuplicate = function(nums) {
    let left = 0, right = nums.length - 1;
    
    while (left <= right) {         // left < right 也可以AC 
        let mid = left + ((right - left) >> 1);
        
        if (nums[mid - 1] == nums[mid]) {
            --mid;

            if ((mid - left) % 2 === 0) {
                left = mid + 2; 
            } else {
                right = mid - 1;
            }
        } else if (nums[mid] === nums[mid + 1]) {
            ++mid;

            if ((right - mid) % 2 === 0) {
                right = mid - 2;
            } else {
                left = mid + 1;
            }
        } else {
            return nums[mid];
        }
    }

    return nums[left];
};
