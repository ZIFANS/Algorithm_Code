# Python

```py
class Solution:
    def mySqrt(self, x: int) -> int:
        if x < 2:
            return x
        
        left, right = 1, x // 2
        
        while left <= right:
            mid = (left + right) // 2

            if (mid * mid == x):
                return mid
            
            if (mid * mid < x):
                left = mid + 1
            else:
                right = mid - 1

        return right
```


# JavaScript

```js
var mySqrt = function(x) {
    if (x < 2)
        return x;
    
    let left = 1, mid, right = Math.floor(x / 2);

    while (left <= right) {
        mid = Math.floor(left + (right - left) / 2);

        if (mid * mid === x)
            return mid;
        
        if (mid * mid < x)
            left = mid + 1;
        else
            right = mid - 1;
    }

    return right;
};
```
