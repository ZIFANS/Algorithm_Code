// 第一想法的代码
var removeElements = function(head, val) {
    let dummy = new ListNode(-1);
    dummy.next = head;
    let pre = dummy, p = dummy.next;

    while (p) {
        if (p.val === val) {
            let temp = p.next;
            pre.next = p.next;
            p = temp;
        } else {
            pre = p;
            p = p.next;
        }
    }

    return dummy.next;
};
