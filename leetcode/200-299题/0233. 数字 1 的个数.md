```python
class Solution:
    def countDigitOne(self, n: int) -> int:
        digit, ans = 1, 0
        high, cur, low = n // 10, n % 10, 0

        while high != 0 or cur != 0:
            if cur == 0:
                ans += high * digit
            elif cur == 1:
                ans += high * digit + low + 1
            else:
                ans += (high + 1) * digit
            
            low += cur *digit
            cur = high % 10
            high //= 10
            digit *= 10
        
        return ans
```

```js
function numOne(num) {
    let ans = 0;

    while (num) {
        let temp = num % 10;

        if (temp === 1)
            ++ans;
        num = Math.floor(num / 10);
    }

    return ans;
}
```

```C++
class Solution {
public:
    int NumberOf1Between1AndN_Solution(int n) {
        int number = 0;
        for(int i = 1; i <= n; ++i) {
            int temp = i;
            while(temp) {
                if(temp % 10 == 1)
                    number++;
                temp = temp / 10;
            }
        }
        return number;
    }
};
```
