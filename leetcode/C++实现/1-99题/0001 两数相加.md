# Python 

```python
# 1、循环暴力解法
class Solution:
    def twoSum(self, nums: List[int], target: int) -> List[int]:
        len1 = len(nums)
        ans = []

        for i in range(len1):
            for j in range(i + 1, len1):
                if nums[i] + nums[j] == target:
                    ans.append(i)
                    ans.append(j)
        

        return ans
```


# JavaScript

```js
// 暴力法
var twoSum = function(nums, target) {
    let len = nums.length;
    let ans = [];
    for(let i = 0; i < len; ++i) {
        for(let j = i + 1; j < len; ++j) {
            if(nums[i] + nums[j] === target) {
                ans.push(i);
                ans.push(j);
            }
        }
    }
    return ans;
};


// 哈希法 自己一下就AC的
var twoSum = function(nums, target) {
    let map = new Map();

    for (let i = 0; i < nums.length; ++i) {
        if (map.has(target - nums[i])) {
            return [map.get(target - nums[i]), i];
        }
        map.set(nums[i], i);
    }
};
```

class Solution {
public:
    vector<int> twoSum(vector<int>& nums, int target) {
        int len = nums.size();
        vector<int> ans;

        for (int i = 0; i < len; ++i)
        {
            for (int j = i + 1; j < len; ++j)
            {
                if (nums[i]+nums[j] == target )
                {
                    ans.push_back(i);
                    ans.push_back(j);
                }
            }
        }
        
        return ans;
    }
};
