```py
class Solution:
    def intersect(self, nums1: List[int], nums2: List[int]) -> List[int]:
        map1 = {}
        ans = []

        for i in range(len(nums1)):
            if nums1[i] in map1:
                map1[nums1[i]] += 1
            else:
                map1[nums1[i]] = 1
        
        for i in range(len(nums2)):
            if (nums2[i] in map1 and map1[nums2[i]] > 0):
                ans.append(nums2[i])
                map1[nums2[i]] -= 1
                
        return ans
```


# JavaScript

```js
var intersect = function(nums1, nums2) {
    let map1 = new Map();
    let ans = [];

    for (let i = 0; i < nums1.length; ++i) {
        if (!map1.has(nums1[i])) {
            map1.set(nums1[i], 1);
        } else {
            let temp = map1.get(nums1[i]) + 1;
            map1.set(nums1[i], temp);
        }
    }

    for (let i = 0; i < nums2.length; ++i) {
        if (map1.get(nums2[i]) > 0) {
            ans.push(nums2[i]);
            let temp = map1.get(nums2[i]) - 1;
            map1.set(nums2[i], temp);
        }
    }

    return ans;
};
```
